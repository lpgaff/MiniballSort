27 std::shared_ptr<FebexData> febex;
28 MiniballDataPackets *data = new MiniballDataPackets;
29 t->SetBranchAddress( "data", &data );

31 // Canvas
32 TCanvas *c1 = new TCanvas( "c1", filename.data(), 900, 1000 );
33 TCanvas *c2 = new TCanvas( "c2", filename.data(), 600, 400 );

35 TGraph *g1, *g2, *g3, *g4, *g5, *g6;
36 std::string title;
37
38 // Energy histogram
39 TH1F *h = new TH1F( "mwd_energy", "Energy spectrum", 65536, -0.5, 65535.5 );

41 // Loop
42 for( unsigned long long i = 0; i < nentries; ++i ){

44 // Get entry
45 t->GetEntry(i);
46
47 // Check if it is FebexData
48 if( !data->IsFebex() ) continue;

50 // If it is febex, get the data packet

53 // Check if it matches the channel we want
54 if( febex->GetSfp() == sfp &&
55 febex->GetBoard() == board &&
56 febex->GetChannel() == ch ) {

61 // Fill histogram

65 // Draw trace - graph1

79 // Draw stage2 - graph3

84 g3->Draw("ac");

115 // CFD

117 arr[k*3].SetLineColor(kRed+1);

121
122 // Energy and baseline
123 c1->cd(5);
124 arr[k*3+1].SetLineColor(kRed+1);
125 arr[k*3+1].DrawArrow( baseline_time, g5->Eval(sampling_time),
126 baseline_time, g5->Eval(baseline_time),
127 0.01, "-|>" );
128 arr[k*3+2].SetLineColor(kRed+1);
129 arr[k*3+2].DrawArrow( sampling_time, g5->Eval(baseline_time),
130 sampling_time, g5->Eval(sampling_time),
131 0.01, "-|>" );
132
133
134 }
135
136 // Draw energy histogram - graph6
137 c2->cd();
138 if( h->Integral() > 50 )
139 h->GetXaxis()->SetRangeUser(
140 h->GetMean() - 10.* h->GetStdDev(),
141 h->GetMean() + 10.* h->GetStdDev() );
